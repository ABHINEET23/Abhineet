import java.util.*;  
public class ThirdLargestInArrayExample1{  
public static int getThirdLargest(int[] a, int total){  
Arrays.sort(a);  
return a[total-3];  
}  
public static void main(String args[]){  
int a[]={1,2,5,6,3,2};  
System.out.println("Third Largest: "+getThirdLargest(a,6));  
}}
